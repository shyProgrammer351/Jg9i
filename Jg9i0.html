열거형

enum   변수에 저장할 수 있는 상수값에 의미를 부여하여 열거한 자료형

enum 열거형이름 { 상수값1, ..., 상수값n };

enum Note { DO, RE, MI, FA, SOL, LA, TI };
Note sound = Note.FA;

enum Color { RED, ORANGE, YELLOW, GREEN, BLUE, INDIGO, PURPLE };
Color color = Color.GREEN;

enum RPSGame { SCISSOR, ROCK, PAPER };
RPSGame player = RPSGame.SCISSOR;

import java.util.Scanner;
public class Direction
{
  enum Direction { left, right, up, down };
  public static void main( String[] args ) {
    Scanner scan = new Scanner( System.in );
    System.out.println( " / " );
    System.out.println( " [ 옷 문" );
    System.out.println( " -> \n" );
    System.out.print("left, right, up, down을 입력하세요:");
    Direction move = Direction.valueOf( scan.next() );
    System.out.print( "\033[2;6f " );
    switch( move ) {
      case up:
        System.out.print( "\033[1;6f 옷/" );
        break;
      case down:
        System.out.print( "\033[3;6f 옷->" );
        break;
      case left:
        System.out.print( "\033[2;1f 옷]" );
        break;
      case right:
        System.out.print( "\033[2J\033[4;15fSUCCESS!" );
        break;
    }
    scan.close();
  }
} 		import java.util.Scanner;

제네릭

T   제네릭 타입으로 미지의 자료형을 나타내며 객체 인스턴스 생성시 자료형이 구체화

extends   제네릭 타입에서 클래스나 인터페이스의 상속관계를 표현

Comparable   두 값을 비교하는 compareTo 메서드를 제공하는 인터페이스

public class GenericSort
{
 public static<T extends Comparable<T>>void sort(T[] list){
    for ( int last = list.length - 1; last >= 0; last-- ) {
      for ( int index = 0; index < last; index++ ) {
        if ( list[index].compareTo( list[index+1] ) > 0 ) {
          T temp = list[ index ];
          list[index] = list[index+1];
          list[index+1] = temp;
        }
      }
    }
  }
